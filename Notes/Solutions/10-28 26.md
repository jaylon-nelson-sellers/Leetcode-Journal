 [[26. Remove Duplicates from Sorted Array]]
```
class Solution:

    def removeDuplicates(self, nums: List[int]) -> int:

        size = len(nums)

        k = 0

        for i in range(size):

            if nums[i - 1] != nums[i]:

                # Updating insertIndex in our main array

                nums[k] = nums[i]

                # Incrementing insertIndex count by 1

                k = k + 1

        return k
```
