```
from typing import List

class Solution:
    def removeElement(self, nums: List[int], target_value: int) -> int:
        """
        Removes all instances of target_value in nums in-place.
        Returns the count of elements not equal to target_value.
        """
        current_index = 0
        active_length = len(nums)

        while current_index < active_length:
            if nums[current_index] == target_value:
                # Replace the target with the last valid element in the array
                # This 'deletes' the target without needing to shift all elements
                last_index = active_length - 1
                nums[current_index] = nums[last_index]
                
                # Shrink the logical size of the array
                active_length -= 1
                
                # Note: We do NOT increment current_index here because the 
                # new element swapped from the back must be checked in the next iteration.
            else:
                # Element is fine, move to the next one
                current_index += 1
                
        return active_length
```
